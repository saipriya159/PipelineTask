pipeline {
    agent any

    stages {
        stage('Build and Run HelloWorld') {
            steps {
                script {
                    echo "Compiling and running HelloWorld.java"
                    sh '''
                    javac HelloWorld.java
                    java HelloWorld
                    '''
                }
            }
        }

        stage('Build and Run HelloWipro') {
            steps {
                script {
                    echo "Compiling and running HelloWipro.java"
                    sh '''
                    javac HelloWipro.java
                    java HelloWipro
                    '''
                }
            }
        }

        stage('Build and Run HelloJenkins') {
            steps {
                script {
                    echo "Compiling and running HelloJenkins.java"
                    sh '''
                    javac HelloJenkins.java
                    java HelloJenkins
                    '''
                }
            }
        }
    }

    post {
        always {
            echo "Pipeline execution completed!"
        }
        success {
            echo "Pipeline completed successfully!"
        }
        failure {
            echo "Pipeline failed!"
        }
    }
}
